### 一键启动指定分辨率的远程桌面文件——办公台式机.rdp
用文本编辑器打开 `.rdp` 文件，添加以下参数：

```bash
screen mode id:i:1
desktopwidth:i:1920
desktopheight:i:1080
session bpp:i:24
```

| 参数                       | 格式与值         | 作用                               | 典型选项                                                     |
| -------------------------- | ---------------- | ---------------------------------- | ------------------------------------------------------------ |
| **`screen mode id:i:1`**   | `i:1` 或 `i:2`   | 控制远程窗口的显示模式             | `1`=全屏<br>`2`=窗口化                                       |
| **`desktopwidth:i:1920`**  | `i:[数值]`       | 设置远程桌面水平像素（分辨率宽度） | `1920` (1080p)<br>`2560` (2K)<br>`3840` (4K)                 |
| **`desktopheight:i:1080`** | `i:[数值]`       | 设置远程桌面垂直像素（分辨率高度） | `1080` (1080p)<br>`1440` (2K)<br>`2160` (4K)                 |
| **`session bpp:i:32`**     | `i:16`/`24`/`32` | 定义颜色深度（位/像素）            | `16`=增强色（65536色）<br>`24`=真彩色（千万色）<br>`32`=带透明通道的真彩色 |

```
@echo off

ipconfig /flushdns


echo.
echo ===== 2.清空本机hosts解析 =====
echo.

    echo 备份文件已存在

    copy /Y C:\Windows\System32\drivers\etc\hosts C:\Windows\System32\drivers\etc\hosts.zdns.bak

echo.
echo ===== 执行完成，10秒后自动退出 =====

```

### 切换DNS.bat
```
@echo off
echo --------可供选择的DNS地址--------
echo 1.办公网络
echo 2.内部应急网络
echo 3.自动获取
set /p input=请输入要设置的DNS地址类型：
if /i "%input%"=="1" goto A
if "%input%"=="2" goto B
if "%input%"=="3" goto C
exit

:A
netsh interface ip set dns "以太网" static 223.5.5.5 >nul
netsh interface ip add dns "以太网" 223.6.6.6 >nul
ipconfig/flushdns
echo 成功将DNS地址设置为办公DNS！
goto end

:B
netsh interface ip set dns "以太网" static 10.201.2.251 >nul
netsh interface ip add dns "以太网" 10.201.2.254 >nul
ipconfig/flushdns
echo 成功将DNS地址设置为内网应急DNS！
goto end

:C
netsh interface ip set dns "以太网" dhcp >nul
ipconfig /flushdns
echo 成功将DNS地址设置为DHCP！
goto end

:end
pause
```

#### 一键切换WiFi——wifi_switch.bat
```bash
@echo off
rem 自动切换连接两个无线信号SSID
title 切换无线WIFI
set "ssid1=Redmi Note 10"
set "ssid2=车e兴ETC"

for /f "tokens=1* delims=:" %%a in ('netsh wlan show interfaces ^| find " SSID "') do (
for /f "tokens=*" %%c in ("%%b") do set "ssid=%%c"
)
echo 当前连接的WiFi: "%ssid%"

if /i "%ssid%" equ "%ssid1%" (
netsh wlan disconnect
netsh wlan connect name="%ssid2%"
echo 已切换到 "%ssid2%"
) else (
netsh wlan disconnect
ping 127.0.0.1 -n 5 >nul
netsh wlan connect name="%ssid1%"
echo 已切换到 "%ssid1%"
)
pause
```

#### 删除空文件夹.bat
```bash
@echo off
echo.
echo 正在删除当前目录及子目录中所有的空文件夹，请稍后……
echo --------------------------------
cd.>listnull.txt
for /f "delims=" %%i in ('dir /ad /b /s') do (
dir /b "%%i"|findstr .>nul || echo %%i>>listnull.txt
)
set /a sum=0
for /f %%i in (listnull.txt) do (
echo 成功删除空目录：%%i
rd /q %%i
set /a sum=sum+1
)
echo ---------------------------------
echo 共成功删除%cd%目录及其子目录下%sum%个空文件夹！
echo.
set sum=
del /q listnull.txt >nul
```
